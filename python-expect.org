#+BEGIN_SRC python
#!/bin/env python                                                                                      
# -*- coding: utf-8 -*-                                                                                
                                                                                                       
import sys                                                                                             
import pexpect                                                                                         
                                                                                                       
if __name__ == "__main__":                                                                             
  if len(sys.argv) != 3:                                                                               
    print "Illegal param number"                                                                       
    sys.exit(1)                                                                                        
                                                                                                       
  user = "zhongyi01"                                                                                   
  passwd = "123456"
  hostname = sys.argv[1]                                                                               
  cmd = sys.argv[2]                                                                                    
                                                                                                       
  print hostname,                                                                                      
  ssh = pexpect.spawn("ssh %s@%s %s" % (user, hostname, cmd))                                          
  try:                                                                                                 
    ret = ssh.expect(['password:', 'continue connecting (yes/no)?'], timeout = 5)                      
    if ret == 0:                                                                                       
      ssh.sendline(passwd)                                                                             
    elif ret == 1:                                                                                     
      ssh.sendline('yes\n')                                                                            
      ssh.expect('password: ')                                                                         
      ssh.sendline(passwd)                                                                             
    ssh.sendline(cmd)                                                                                  
    r = ssh.read()                                                                                     
    print r                                                                                            
  except pexpect.EOF:                                                                                  
    print "EOF"                                                                                        
  except pexpect.TIMEOUT:                                                                              
    print "TIMEOUT"                                                                                    
                                                                                                       
  ssh.close()                   
#+END_SRC
